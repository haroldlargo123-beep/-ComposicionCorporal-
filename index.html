// Devuelve JSON normal o JSONP según el parámetro ?callback=
// Esto resuelve el bloqueo de CORS desde GitHub Pages / incógnito.
function respond(data, callback) {
  var json = JSON.stringify(data);
  if (callback) {
    return ContentService
      .createTextOutput(callback + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var d = JSON.parse(e.postData.contents);
    var s = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    if (s.getLastRow() === 0) s.appendRow(HEADERS);

    if (d.action === 'update' && d._row) {
      var row = d._row;
      s.getRange(row, 1).setValue(d.fecha||'');
      s.getRange(row, 2).setValue(d.nombre||'');
      s.getRange(row, 3).setValue(d.sexo||'');
      s.getRange(row, 4).setValue(d.edad||'');
      return respond({status:"ok"}, null);
    }

    if (d.action === 'delete' && d._row) {
      s.deleteRow(d._row);
      return respond({status:"ok"}, null);
    }

    s.appendRow([
      d.fecha, d.nombre, d.sexo, d.edad,
      d.an_peso_t1, d.an_peso_t2, d.peso,
      d.an_estatura_t1, d.an_estatura_t2, d.estatura,
      d.pl_subescapular_t1, d.pl_subescapular_t2,
      d.pl_tricipital_t1, d.pl_tricipital_t2,
      d.pl_musloAnterior_t1, d.pl_musloAnterior_t2,
      d.pl_pierna_t1, d.pl_pierna_t2,
      d.pl_suprailiaco_t1, d.pl_suprailiaco_t2,
      d.pl_abdominal_t1, d.pl_abdominal_t2,
      d.pl_supraespinal_t1, d.pl_supraespinal_t2,
      d.an_dEstiloideo_t1, d.an_dEstiloideo_t2,
      d.an_dBicondiloFemur_t1, d.an_dBicondiloFemur_t2,
      d.an_dBiepicondilarH_t1, d.an_dBiepicondilarH_t2,
      d.an_pBrazo_t1, d.an_pBrazo_t2,
      d.an_pPierna_t1, d.an_pPierna_t2,
      d.grasaFaulkner, d.grasaYuhasz, d.pesoGraso,
      d.masaOsea, d.masaResidual, d.pesoMuscular,
      d.endomorphy, d.mesomorphy, d.ectomorphy,
      d.X, d.Y, d.classification
    ]);
    return respond({status:"ok"}, null);
  } catch(err) {
    return respond({status:"error",message:err.message}, null);
  }
}

function doGet(e) {
  var action   = e && e.parameter && e.parameter.action;
  var callback = e && e.parameter && e.parameter.callback; // JSONP callback
  var s = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();

  if (action === 'ping') {
    return respond({status:"ok",message:"Conectado"}, callback);
  }

  var rows = s.getDataRange().getValues();
  if (rows.length <= 1)
    return respond({status:"ok",data:[]}, callback);

  var headers = rows[0];
  var data = rows.slice(1).map(function(r,i){
    var obj = {_row: i+2};
    headers.forEach(function(h,j){ obj[h] = r[j]; });
    return obj;
  });
  return respond({status:"ok",data:data}, callback);
}
